def plot_reasons(df, cleanup=False, percent=False):
    if cleanup:
        # Handle missing values
        # Drop rows where 'Year' is missing
        df = df.dropna(subset=['Year'])
        # Convert 'Year' values to lowercase
        df['Year'] = df['Year'].str.lower()

    # Exclude the 'Lines_of_code' column from the data to be summed
    data_excluded = df.drop('Lines_of_code', axis=1)

    # Group the modified data by 'Year' and sum the values of the remaining columns
    sum_data = data_excluded.groupby('Year').sum()

    if percent:
        # Calculate the total responses per 'Year' and divide each category by this total to get percentages
        sum_data = sum_data.div(sum_data.sum(axis=1), axis=0) * 100

    # Define the order of the 'Year' categories
    year_order = ['freshman', 'sophomore', 'junior', 'senior', 'g_nonmba', 'mba']

    # Reindex the DataFrame according to the specified order
    sum_data = sum_data.reindex(year_order)

    # Create the bar plot
    sum_data.plot(kind='bar', stacked=False)  # Use stacked=False for separate bars for each column
    if cleanup:
        plt.title('Sum of Reasons by Year Group CLEAN')
    else:
        plt.title('Sum of Reasons by Year Group NOT CLEAN')

    plt.ylabel('Percentage of Responses' if percent else 'Total Responses')
    plt.xlabel('Year')
    plt.xticks(rotation=45)  # Rotating the x-axis labels for better readability
    plt.tight_layout()  # Adjusts the plot to ensure everything fits without overlap

def plot_lines_of_code(df,drop_missing=True):
    # Clean data
    if drop_missing:
        # Drop rows where either 'Year' or 'Lines_of_code' is missing
        df = df.dropna(subset=['Year', 'Lines_of_code'])
    else:
        # Drop rows where 'Year' is missing
        df = df.dropna(subset=['Year'])
        # Handle missing values in 'Lines_of_code' by filling them with the mean of the column
        df['Lines_of_code'] = df['Lines_of_code'].fillna(df['Lines_of_code'].mean())
        
    # Convert 'Year' values to lowercase
    df['Year'] = df['Year'].str.lower()

    # Group by 'Year' and calculate the average 'Lines_of_code'
    avg_lines = df.groupby('Year')['Lines_of_code'].mean()

    # Define the order of the 'Year' categories
    year_order = ['freshman', 'sophomore', 'junior', 'senior', 'g_nonmba', 'mba']
    
    # Reindex the data according to the specified year order to ensure the plot follows this order
    avg_lines = avg_lines.reindex(year_order)

    # Create the bar plot
    avg_lines.plot(kind='bar')
    plt.title('Average Lines of Code by Year Group')
    plt.ylabel('Average Lines of Code')
    plt.xlabel('Year')
    plt.xticks(rotation=45)  # Rotating the x-axis labels for better readability
    plt.tight_layout()  # Adjusts the plot to ensure everything fits without overlap

def compute_correlations(df):
    # Drop rows with any missing data
    df_clean = df.dropna()
    
    # Convert all string columns to lowercase
    # Explicitly convert all string columns to lowercase using .loc
    str_columns = df_clean.select_dtypes(include=['object']).columns
    df_clean.loc[:, str_columns] = df_clean[str_columns].apply(lambda x: x.str.lower())
    
    # Define a dictionary to store correlation data for each 'Year'
    correlations = {}
    
    # Loop through each unique 'Year'
    for year in df_clean['Year'].unique():
        # Filter the DataFrame by 'Year'
        year_data = df_clean[df_clean['Year'] == year]
        
        # Calculate the correlation between 'Lines_of_code' and other columns
        correlation = year_data.corr()['Lines_of_code'].drop('Lines_of_code')  # drop self-correlation
        
        # Store the correlation data in the dictionary
        correlations[year] = correlation
    
    # Print the correlations for inspection
    for year, corr in correlations.items():
        print(f"Correlations for {year}:")
        print(corr, "\n")

def compareAL_plot_lines_of_code(filelist, drop_missing=True):
    fig, ax = plt.subplots(figsize=(12, 6))  # Adjusted figure size for clarity
    colors = ['blue','orange','skyblue', 'khaki']  # Define colors for differentiation
    num_datasets = len(filelist)
    width = 0.8 / num_datasets  # Adjusting width dynamically based on the number of datasets
    years_labels = ['freshman', 'sophomore', 'junior', 'senior', 'g_nonmba', 'mba']  # Year categories
    gap_width = 0.1  # Define the width of the gap between the first and second sets

    for index, df in enumerate(filelist):
        # Clean data
        if drop_missing:
            df = df.dropna(subset=['Year', 'Lines_of_code'])
        else:
            df = df.dropna(subset=['Year'])
            df['Lines_of_code'] = df['Lines_of_code'].fillna(df['Lines_of_code'].mean())

        # Convert 'Year' values to lowercase
        df['Year'] = df['Year'].str.lower()

        # Group by 'Year' and calculate the average 'Lines_of_code'
        avg_lines = df.groupby('Year')['Lines_of_code'].mean()

        # Reindex the data according to the specified year order
        avg_lines = avg_lines.reindex(years_labels)

        # Calculate bar positions
        if index >= 2:  # Start adding gap after the second dataset
            positions = np.arange(len(years_labels)) + index * width + gap_width
        else:
            positions = np.arange(len(years_labels)) + index * width

        # Plot each dataset
        course_labels = ['100A fa23', '100A sp24', '100L fa24', '100L sp24']
        ax.bar(positions, avg_lines, color=colors[index],width=width, label=course_labels[index])

    # Add vertical dashed lines to separate the years
    for pos in np.arange(len(years_labels)) - 0.5:
        ax.axvline(x=pos + width * num_datasets / 2 - gap_width / 2, color='gray', linestyle='--', linewidth=1)

    # Set plot details
    ax.set_title('Average Lines of Code by Year Group')
    ax.set_ylabel('Average Lines of Code')
    ax.set_xlabel('Year')
    ax.set_xticks(np.arange(len(years_labels)) + (width * num_datasets / 2 - width / 2) + gap_width / 2)
    ax.set_xticklabels(years_labels, rotation=45)
    ax.legend()  # Add a legend to identify the datasets

    plt.tight_layout()  # Adjust layout
    plt.show()

def compareAL_correlations(filelist, drop_missing=True):
    dataset_correlations = {}
    dataset_labels = ['100A fa23', '100A sp24', '100L fa23', '100L sp24']  # Labels for each dataset
    years_labels = ['freshman', 'sophomore', 'junior', 'senior', 'g_nonmba', 'mba']  # Desired year order

    fig, axes = plt.subplots(2, 2, figsize=(12, 8))  # Create a 2x2 subplot grid
    axes = axes.flatten()  # Flatten the axes array for easy iteration

    for index, df in enumerate(filelist):
        if drop_missing:
            df_clean = df.dropna().copy()
        else:
            df_clean = df.dropna(subset=['Year'])
            df_clean['Lines_of_code'] = df_clean['Lines_of_code'].fillna(df_clean['Lines_of_code'].mean())
            df_clean = df_clean.copy()

        # Convert all string columns to lowercase
        str_columns = df_clean.select_dtypes(include=['object']).columns
        df_clean.loc[:, str_columns] = df_clean.loc[:, str_columns].apply(lambda x: x.str.lower())

        correlations_dict = {}

        # Loop through each unique 'Year'
        for year in df_clean['Year'].unique():
            year_data = df_clean[df_clean['Year'] == year]
            # Calculate the correlation between 'Lines_of_code' and other columns
            correlation = year_data.corr(numeric_only=True)['Lines_of_code'].drop('Lines_of_code')
            correlations_dict[year] = correlation

        # Convert dictionary of correlations to DataFrame and reindex
        correlation_df = pd.DataFrame(correlations_dict).reindex(columns=years_labels)
        dataset_correlations[dataset_labels[index]] = correlation_df

        # Plot heatmap in the corresponding subplot
        sns.heatmap(correlation_df, annot=True, cmap='coolwarm', fmt=".2f", ax=axes[index])
        axes[index].set_title(f'Correlation with Lines of Code Heatmap for {dataset_labels[index]}')

    for label, matrix in dataset_correlations.items():
        print(f"Correlation matrix for {label}:")
        print(matrix, "\n")

    plt.tight_layout()  # Adjust layout to prevent overlap
